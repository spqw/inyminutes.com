01 What is Python?

Python is a dynamically-typed, interpreted language known for its clean syntax and massive ecosystem. It's the #1 language for data science, machine learning, scripting, and a top choice for web backends.

  • Readable — indentation-based syntax enforces clean code.
  • Batteries included — rich standard library for files, HTTP, JSON, regex, and more.
  • Huge ecosystem — 500k+ packages on PyPI.
# Python reads like pseudocode
for name in ["Alice", "Bob", "Charlie"]:
    if len(name) > 3:
        print(f"Hello, {name}!")

03 Variables & Data Types

Python is dynamically typed — you don't declare types, the interpreter infers them.

# Strings
name = "Alice"
greeting = f"Hello, {name}!"     # f-strings (Python 3.6+)
multiline = """Line one
Line two"""

# Numbers
age = 30                  # int (arbitrary precision)
price = 19.99             # float
big = 1_000_000           # underscores for readability

# Booleans & None
is_active = True
is_admin = False
result = None             # Python's null

# Type checking
type(name)                # <class 'str'>
isinstance(age, int)    # True

Type Conversions

int("42")        # 42
float("3.14")    # 3.14
str(100)         # "100"
bool(0)          # False  (0, "", [], None are falsy)
list("abc")      # ["a", "b", "c"]

04 Collections

Lists (mutable, ordered)

fruits = ["apple", "banana", "cherry"]
fruits.append("date")        # Add to end
fruits.insert(0, "avocado") # Insert at index
fruits.pop()                  # Remove & return last

# Slicing: list[start:stop:step]
fruits[1:3]                  # ["banana", "cherry"]
fruits[::-1]                  # Reversed copy

# Useful methods
len(fruits)                   # Length
sorted(fruits)                # New sorted list
"apple" in fruits            # True (membership test)

Dictionaries (mutable, key-value pairs)

user = {
    "name": "Alice",
    "age": 30,
    "email": "alice@example.com"
}

user["name"]                 # "Alice"
user.get("phone", "N/A")    # "N/A" (default if missing)
user["role"] = "admin"       # Add/update key
del user["email"]             # Remove key

# Iteration
for key, value in user.items():
    print(f"{key}: {value}")

Tuples & Sets

# Tuples (immutable)
point = (10, 20)
x, y = point                 # Unpacking

# Sets (unique, unordered)
tags = {"python", "web", "api"}
tags.add("flask")
tags.discard("web")

a = {1, 2, 3}
b = {2, 3, 4}
a | b                         # {1, 2, 3, 4}  union
a & b                         # {2, 3}        intersection
a - b                         # {1}           difference

05 Control Flow

# if / elif / else
score = 85
if score >= 90:
    grade = "A"
elif score >= 80:
    grade = "B"
else:
    grade = "C"

# for loops
for i in range(5):            # 0, 1, 2, 3, 4
    print(i)

for i, fruit in enumerate(fruits):
    print(f"{i}: {fruit}")

# while loops
count = 0
while count < 5:
    count += 1

# Ternary expression
status = "adult" if age >= 18 else "minor"

# Match statement (Python 3.10+)
match command:
    case "quit":
        exit()
    case "hello":
        print("Hi!")
    case _:
        print("Unknown")

06 Functions

# Basic function
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

greet("Alice")                   # "Hello, Alice!"
greet("Bob", greeting="Hey")   # "Hey, Bob!"

# *args and **kwargs
def log(*args, **kwargs):
    print("Args:", args)          # Tuple of positional args
    print("Kwargs:", kwargs)      # Dict of keyword args

log(1, 2, key="value")

# Lambda (anonymous functions)
double = lambda x: x * 2
sorted(users, key=lambda u: u["age"])

# Type hints (optional but recommended)
def add(a: int, b: int) -> int:
    return a + b

Decorators

import time

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"{func.__name__} took {time.time()-start:.2f}s")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    return "done"

07 Classes & OOP

class User:
    def __init__(self, name: str, age: int):
        self.name = name
        self.age = age

    def greet(self) -> str:
        return f"Hi, I'm {self.name}"

    def __repr__(self):
        return f"User({self.name!r}, {self.age})"

alice = User("Alice", 30)
alice.greet()   # "Hi, I'm Alice"

Inheritance & Properties

class Admin(User):
    def __init__(self, name, age, permissions):
        super().__init__(name, age)
        self._permissions = permissions

    @property
    def permissions(self):
        return self._permissions.copy()

admin = Admin("Bob", 35, ["read", "write"])
admin.permissions   # ["read", "write"]

08 Comprehensions & Generators

# List comprehension
squares = [x**2 for x in range(10)]
evens = [x for x in range(20) if x % 2 == 0]

# Dict comprehension
word_lengths = {w: len(w) for w in ["hello", "world"]}
# {"hello": 5, "world": 5}

# Set comprehension
unique_lengths = {len(w) for w in ["hi", "hey", "ho"]}

# Generator expression (lazy, memory-efficient)
total = sum(x**2 for x in range(1_000_000))

# Generator function
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

fib = fibonacci()
[next(fib) for _ in range(8)]  # [0, 1, 1, 2, 3, 5, 8, 13]

09 Error Handling

try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
except (TypeError, ValueError):
    print("Type or value error")
else:
    print("No errors!")         # Runs if no exception
finally:
    print("Always runs")

# Custom exceptions
class NotFoundError(Exception):
    pass

def find_user(user_id):
    if user_id not in database:
        raise NotFoundError(f"User {user_id} not found")

# Context managers (auto-cleanup)
with open("data.txt", "r") as f:
    content = f.read()
# File is automatically closed here
