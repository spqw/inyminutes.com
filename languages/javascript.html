01 What is JavaScript?

JavaScript is the language of the web. It runs in every browser, on servers via Node.js, in mobile apps, and even desktop applications. It's dynamically typed, prototype-based, and supports both functional and object-oriented styles.

  • Universal — the only language that runs natively in browsers.
  • Event-driven — non-blocking I/O with async/await.
  • Huge ecosystem — npm has 2M+ packages.

03 Variables & Types

// Use const by default, let when you need to reassign
const name = "Alice";         // Cannot be reassigned
let age = 30;                 // Can be reassigned
// var is legacy — avoid it (function-scoped, hoisted)

// Primitive types
const str = "hello";           // string
const num = 42;               // number (int & float are same)
const big = 9007199254740993n; // bigint
const bool = true;            // boolean
const nothing = null;         // null (intentional absence)
let x;                        // undefined (not yet assigned)
const sym = Symbol("id");    // symbol (unique identifier)

// typeof
typeof "hello"   // "string"
typeof 42        // "number"
typeof null      // "object" (famous JS bug)

04 Strings & Template Literals

// Template literals (backticks)
const greeting = `Hello, ${name}! You are ${age} years old.`;

// Multi-line strings
const html = `
  <div class="card">
    <h2>${name}</h2>
  </div>
`;

// Useful methods
"hello".toUpperCase()        // "HELLO"
"hello world".split(" ")    // ["hello", "world"]
"hello".includes("ell")     // true
"  hello  ".trim()           // "hello"
"hello".padStart(8, ".")    // "...hello"

05 Arrays & Objects

Arrays

const fruits = ["apple", "banana", "cherry"];

fruits.push("date");           // Add to end
fruits.pop();                   // Remove from end
fruits.includes("apple");     // true
fruits.indexOf("banana");     // 1

// Powerful array methods
const nums = [1, 2, 3, 4, 5];
nums.map(n => n * 2)          // [2, 4, 6, 8, 10]
nums.filter(n => n > 2)       // [3, 4, 5]
nums.reduce((sum, n) => sum + n, 0)  // 15
nums.find(n => n > 3)         // 4
nums.some(n => n > 4)         // true
nums.every(n => n > 0)        // true

Objects

const user = {
    name: "Alice",
    age: 30,
    greet() {
        return `Hi, I'm ${this.name}`;
    }
};

user.name                     // "Alice"
user["age"]                   // 30 (bracket notation)
user.email = "a@b.com";       // Add property

// Destructuring
const { name, age } = user;
const { name: userName } = user;  // Rename

// Spread
const updated = { ...user, age: 31 };

06 Control Flow

// if / else
if (score >= 90) {
    grade = "A";
} else if (score >= 80) {
    grade = "B";
} else {
    grade = "C";
}

// Ternary
const status = age >= 18 ? "adult" : "minor";

// for...of (iterate values)
for (const fruit of fruits) {
    console.log(fruit);
}

// for...in (iterate keys — use for objects)
for (const key in user) {
    console.log(`${key}: ${user[key]}`);
}

// Classic for loop
for (let i = 0; i < 5; i++) {
    console.log(i);
}

07 Functions

// Function declaration (hoisted)
function add(a, b) {
    return a + b;
}

// Arrow function (not hoisted, no own 'this')
const multiply = (a, b) => a * b;
const square = x => x * x;          // Single param: no parens needed

// Default parameters
function greet(name = "World") {
    return `Hello, ${name}!`;
}

// Rest parameters
function sum(...nums) {
    return nums.reduce((a, b) => a + b, 0);
}
sum(1, 2, 3);  // 6

// Closures
function counter() {
    let count = 0;
    return () => ++count;
}
const inc = counter();
inc(); // 1
inc(); // 2

08 Async JavaScript

// Promises
const promise = new Promise((resolve, reject) => {
    setTimeout(() => resolve("done!"), 1000);
});

promise.then(result => console.log(result));

// async/await — cleaner syntax for promises
async function fetchUser(id) {
    try {
        const res = await fetch(`/api/users/${id}`);
        const data = await res.json();
        return data;
    } catch (err) {
        console.error("Failed:", err);
    }
}

// Parallel async operations
const [users, posts] = await Promise.all([
    fetch("/api/users").then(r => r.json()),
    fetch("/api/posts").then(r => r.json()),
]);

09 ES6+ Features

// Destructuring
const [first, ...rest] = [1, 2, 3, 4];
const { name, age, ...others } = user;

// Spread operator
const merged = [...arr1, ...arr2];
const clone = { ...original };

// Optional chaining (?.) — stops at null/undefined
const city = user?.address?.city;       // undefined if any part is null
const first = users?.[0]?.name;

// Nullish coalescing (??) — only null/undefined trigger default
const port = config.port ?? 3000;      // 0 would NOT trigger default

// Logical assignment
user.name ??= "Anonymous";             // Assign if null/undefined
user.cache ||= loadCache();           // Assign if falsy

// Modules
import { readFile } from "fs/promises";
export const helper = () => { };
